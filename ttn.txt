DECODER:

function Decoder(bytes, port) {
  var decoded = {
    counter: bytes[0],
    battery: (bytes[2] << 8 | bytes[1])/100,
    temp_ds: (bytes[4] << 8 | bytes[3])/100
  };
  return decoded;
}

Example:
{
  "battery": 35,
  "counter": 1,
  "temp_ds": 0
}


ENCODER:

function Encoder(object, port) {
  var bytes = [];

  bytes[0] = object.sleepTimeSec & 0xff
  bytes[1] = object.sleepTimeSec >> 8

  return bytes;
}

Example:
{
  "sleepTimeSec": 1000
}
